Run the demo in console:
var Object = {};
Object.demo = function(a, b) {
    return a - b;
}

AopUtil.before("Object.demo", function(a, b) {
    alert(0);
    // [a * 2, b] would be injected as agrument list to the next advice (or the original function)
    return [a * 2, b];
}, 2);
AopUtil.after("Object.demo", function(result) {
    // receives 1 param only from the original function.
    alert(1);
    return result * 2;
}, 1);

Object.demo(1,1);
Correct output should be 1.
Enjoy.
